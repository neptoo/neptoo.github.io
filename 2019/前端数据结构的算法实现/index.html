function quickSort(arr) {
    if (arr.length <= 1) {
        return arr;   // é€’å½’å‡ºå£
    }
    var left = [],
        right = [],
        current = arr.splice(0, 1);
    for (let i = 0; i < arr.length; i++) {
        if (arr[i] < current) {
            left.push(arr[i])
        } else {
            right.push(arr[i])
        }
    }
    return quickSort(left).concat(current,quickSort(right));  // é€’å½’
}

function bubbleSort(arr){
  var len = arr.length;
  for(let outer = len;outer >= 2;outer--){
    for(let inner = 0;inner <= outer-1;inner++){
     if(arr[inner] > arr[inner+1]){
       let temp = arr[inner];
       arr[inner] = arr[inner+1];
       arr[inner+1] = temp;
     }
    }
  }
  return arr;
}
